# 如何实现函数的柯里化?比add(1)(2)(3)
  函数的柯里化（Currying）是一种将接受多个参数的函数转换成一系列只接受一个参数的函数的技术。
  对于你提到的 add(1)(2)(3) 这种形式，其核心思想是：
  函数接收第一个参数后，返回一个新的函数。
  这个新函数接收第二个参数，再返回另一个新函数。
  直到所有参数都收集完毕，才执行最终的计算并返回结果。
  柯里化的主要好处是参数复用和函数组合，在函数式编程（如 React Redux, Lodash fp）中非常常见。


// 方法一：手动嵌套返回函数（最基础）
// 适用于参数数量固定的情况（比如固定 3 个参数）。
```js
function add(a) {
  return function(b) {
    return function(c) {
      return a + b + c;
    };
  };
}

console.log(add(1)(2)(3)); // 输出: 6
// 原理：
// add(1) 返回一个等待 b 的函数。
// 该函数传入 2 后，返回一个等待 c 的函数。
// 最后传入 3，执行 a + b + c。
```
// 方法二：通用柯里化函数（支持任意参数数量）
```js
// 如果你想要一个通用的工具函数，可以将任何普通函数转换为柯里化形式，并且支持分步调用或一次性调用。
function curry(fn) {
  return function curried(...args) {
    // 如果参数数量足够，直接执行
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    } else {
      // 否则返回一个新函数，继续收集参数
      return function(...moreArgs) {
        return curried.apply(this, args.concat(moreArgs));
      };
    }
  };
}

// 定义一个普通加法函数
function add(a, b, c) {
  return a + b + c;
}

// 转换为柯里化函数
const curriedAdd = curry(add);

console.log(curriedAdd(1)(2)(3));       // 输出: 6 (分步调用)
console.log(curriedAdd(1, 2)(3));       // 输出: 6 (混合调用)
console.log(curriedAdd(1)(2, 3));       // 输出: 6 (混合调用)
console.log(curriedAdd(1, 2, 3));       // 输出: 6 (一次性调用)
// 关键点：
// fn.length 获取原函数期望的参数个数。
// 利用闭包保存已传入的参数 args。
// 递归返回新函数直到参数凑齐。

```

```js
// 方法三：无限累加（特殊场景：add(1)(2)(3)...()）
// 有时候面试题会要求实现 add(1)(2)(3)... 直到调用无参数时才计算结果。这需要重写函数的 toString 或 valueOf 方法。
function add(a) {
  let sum = a;

  function inner(b) {
    sum += b;
    return inner;
  }

  // 当函数被当作字符串或数字使用时触发
  inner.toString = () => sum;
  inner.valueOf = () => sum;

  return inner;
}

console.log(add(1)(2)(3).toString()); // "6"
console.log(+add(1)(2)(3));           // 6 (隐式转换)
// 注意：如果不转字符串或数字，它始终返回函数本身
```
